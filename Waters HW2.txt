Coli datamkdir ~/workcd ~/workcurl -O -L https://s3.amazonaws.com/public.ged.msu.edu/ecoli_ref-5m.fastq.gzSplit readsgunzip -c ecoli_ref-5m.fastq.gz | split-paired-reads.py -1 top.R1.fq -2 top.R2.fqRun TrimmomaticTrimmomaticPE top.R1.fq top.R2.fq out-r1.fq orphan1 out-r2.fq orphan2 ILLUMINACLIP:TruSeq3-PE.fa:2:40:15 LEADING:2 TRAILING:2 SLIDINGWINDOW:4:2 MINLEN:25Interleave readsinterleave-reads.py out-r1.fq out-r2.fq > out-pe.fqRun Megahit~/megahit/megahit --12 out-pe.fq -o megaoutSave assemblycp megaout/final.contigs.fa lab2-assembly.faUse Quast to measure the assemblycd ~/work~/quast/quast.py lab2-assembly.fa -o lab2_reportpython2.7 ~/quast/quast.py lab2-assembly.fa -o lab2_report
